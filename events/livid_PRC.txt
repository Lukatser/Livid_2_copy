#romance of the three kingdoms
country_event = {
	id = grandmod_china.1
	title = china1t
	desc = china1d
	picture = IMPERIAL_SEAL_eventPicture
	
	trigger = {
		OR = {
			tag = PRC
			tag = ROC
			is_emperor_of_china = yes
		}
		#has_institution = printing_press
		has_country_flag = prc_literature
		OR = {
			war_with = SHU
			war_with = WUU
		}
	}
	mean_time_to_happen = {
		months = 6
	}
	fire_only_once = yes
	
	option = {
		name = china1a
		add_country_modifier = {
			name = "romance_of_the_three_kingdoms" #10% morale, 10% manpower, -10% goods
			duration = 3650
		}
		add_army_tradition = 25
		WUU = {
			every_owned_province = {
				add_claim = ROOT
			}
		}
		SHU = {
			every_owned_province = {
				add_claim = ROOT
			}
		}
		ai_chance = { factor = 100 }
	}
	option = {
		name = china1b
		add_prestige = 25
		ai_chance = { factor = 0 }
	}
}
